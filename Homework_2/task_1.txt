Почему код дает именно такие результаты?

var a = 1, b = 1, c, d;
c = ++a; alert(c); 		// 2 Используется префиксная форма записи
d = b++; alert(d); 		// 1 Используется постфиксная форма записи
c = (2+ ++a); alert(c); 	// 5 Используется префиксная форма записи
d = (2+ b++); alert(d); 	// 4 Используется постфиксная форма записи
alert(a); 			// 3 Дважды было выполнено инкрементирование переменной
alert(b); 			// 3 Дважды было выполнено инкрементирование переменной
